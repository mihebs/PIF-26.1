#include <stdio.h>

int main(){
    float num1, num2, num3;
    
    printf("Olá, usuário! Por favor, digite 3 números para ver o resultado da média aritmética entre eles:");
    
    scanf("%f", &num1);
    scanf("%f", &num2);
    scanf("%f", &num3);
    
    float media = (num1 + num2 + num3) / 3;
    
    printf("A média é: %f", media);
}
