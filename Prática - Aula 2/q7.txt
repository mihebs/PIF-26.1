#include <stdio.h>

int main(){
    int num1, num2;
    
    printf("Olá, usuário! Insira 2 numeros inteiros para ver o resto da divisão entre eles:");
    
    scanf("%d", &num1);
    scanf("%d", &num2);

    float resto = num1 % num2;
    
    printf("O resto da divisão é: %.2f", resto);
}
